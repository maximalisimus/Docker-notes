version: '3.1'
services:

  mariadb:
    image: mariadb:latest
    container_name: mariadb
    restart: unless-stopped
    # hostname: mariadb
    # ports:
      # 3306:3306
    expose: '3306'
    environment:
      MYSQL_ROOT_PASSWORD: 'powerdns'
      MYSQL_DATABASE: 'powerdns'
      MYSQL_USER: 'powerdns'
      MYSQL_PASSWORD: 'powerdns'
    volumes:
      - ./mysql-master:/var/lib/mysql

  pdnsmaster:
    image: pschiffe/pdns-mysql
    container_name: pdnsmaster
    restart: unless-stopped
    hostname ns1.debian.local
    ports:
      - 53:53
      - 53:53/udp
    link:
      - mariadb:mysql
    environment:
      - PDNS_master=yes
      - PDNS_api=yes
      - PDNS_api_key=secret # jdnAJHwFflX2 A8DAhUMgL3Kv slave
      # openssl rand -base64 12 | sed 's/[^0-9.a-zA-Z]*//g'

      - PDNS_webserver_address=0.0.0.0
      - PDNS_webserver_password=secret2
      - PDNS_version_string=anonymous
      - PDNS_default_ttl=1500
      - PDNS_soa_minimum_ttl=1200
      - PDNS_default_soa_name=ns1.debian.local
      - PDNS_default_soa_mail=hostmaster.debian.local
      # PDNS_allow_axfr_ips=192.168.100.45
      # PDNS_only_notify=192.168.100.45
      # webserver-allow-from=172.5.0.0/16
      - PDNS_gmysql_host=mariadb
      - PDNS_gmysql_port=3306
      - PDNS_gmysql_user='powerdns'
      - PDNS_gmysql_password='powerdns'
      - PDNS_gmysql_dbname='powerdns'
  
  mdbuwsgi:
    image: mariadb:latest
    container_name: mdbuwsgi
    restart: unless-stopped
    # hostname: mdbuwsgi
    # ports:

    expose: '3310'

      MYSQL_ROOT_PASSWORD: 'root'
      MYSQL_DATABASE: 'powerdnsadmin'
      MYSQL_USER: 'root'
      MYSQL_PASSWORD: 'powerdnsadmin'
    volumes:
      - ./mysql-uwsgi:/var/lib/mysql

  pdnsuwsgi:
    image: pschiffe/pdns-admin-uwsgi:ngoduykhanh
    container_name: pdnsuwsgi
    restart: unless-stopped
    environment:
      - PDNS_ADMIN_SQLA_DB_HOST='mdbuwsgi'
      - PDNS_ADMIN_SQLA_DB_PORT='3310'
      - PDNS_ADMIN_SQLA_DB_USER='root'
      - PDNS_ADMIN_SQLA_DB_PASSWORD='powerdnsadmin'
      - PDNS_ADMIN_SQLA_DB_NAME='powerdnsadmin'     
    link:

  pdnsstatic:
