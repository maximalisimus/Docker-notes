docker-compose.yml

version: '3.1'
services:

  pdns-recursor:
    image: pschiffe/pdns-recursor
    networks:
      - pdns
    volumes:
      - /etc/localtime:/etc/localtime:ro

  mariadb:
    image: mariadb:10.6
    networks:
      pdns:
        aliases:
          - db
          - mysql
    volumes:
      - /etc/localtime:/etc/localtime:ro
      - ./mariadb:/var/lib/mysql:z
    environment:
      - MYSQL_ROOT_PASSWORD=my-secret-pw
    healthcheck:
      test: ['CMD', 'mysqladmin', 'ping', '-h', 'localhost']
      timeout: 10s
      retries: 5

  phpmyadmin:
    image: phpmyadmin/phpmyadmin:5
    networks:
      - pdns
    ports:
      - '8988:80'
    volumes:
      - /etc/localtime:/etc/localtime:ro

  unbound:
    image: "mvance/unbound:latest"
    container_name: unbound
    restart: unless-stopped
    hostname: "unbound"
    volumes:
      - "./unbound:/opt/unbound/etc/unbound/"
    networks:
      pdns:
        ipv4_address: 172.6.0.22
        aliases:
          - unbound

  pdns-master:
    image: pschiffe/pdns-mysql
    container_name: pdnsmaster
    restart: unless-stopped
    hostname: ns1.example.com
    networks:
      pdns:
        ipv4_address: 172.6.0.20
        aliases:
          - pdns
          - unbound
    extra_hosts:
      - 'ns1.example.com:172.6.0.20'
      - 'ns2.example.com:172.6.0.21'
    volumes:
      - /etc/localtime:/etc/localtime:ro
    ports:
      - 53:53
      - 53:53/udp
      - 8081:8081
    dns:
      - 172.6.0.22
      - 8.8.8.8
      - 8.8.4.4
    environment:
      - PDNS_gmysql_password=my-secret-pw
      - PDNS_master=yes
      - PDNS_api=yes
      - PDNS_api_key=secret
      - PDNS_webserver=yes
      - PDNS_webserver_address=127.0.0.1
      - PDNS_webserver_port=8081
      - PDNS_webserver_allow_from=0.0.0.0/0,::/0
      - PDNS_version_string=anonymous
      - PDNS_default_ttl=1500
      - PDNS_allow_axfr_ips=172.6.0.21
      - PDNS_only_notify=172.6.0.21
    depends_on:
      - mariadb

  pdns-slave:
    image: pschiffe/pdns-mysql
    hostname: ns2.example.com
    container_name: pdns-slave
    networks:
      pdns:
        ipv4_address: 172.6.0.21
    extra_hosts:
      - 'ns1.example.com:172.6.0.20'
      - 'ns2.example.com:172.6.0.21'
    volumes:
      - /etc/localtime:/etc/localtime:ro
    environment:
      - PDNS_gmysql_dbname=powerdnsslave
      - PDNS_gmysql_password=my-secret-pw
      - PDNS_slave=yes
      - PDNS_superslave=yes
      - PDNS_webserver=yes
      - PDNS_webserver_address=0.0.0.0
      - PDNS_webserver_allow_from=172.6.0.0/16
      - PDNS_version_string=anonymous
      - PDNS_disable_axfr=yes
      - PDNS_allow_notify_from=172.6.0.20
      - SUPERMASTER_IPS=172.6.0.20
    depends_on:
      - mariadb
      - pdns-master

  pdns-admin-uwsgi:
    image: pschiffe/pdns-admin-uwsgi
    container_name: pdns-admin-uwsgi
    restart: unless-stopped
    environment:
      - PDNS_ADMIN_SQLA_DB_PASSWORD='my-secret-pw'
      - PDNS_VERSION=4.5
      - PDNS_API_KEY=secret
    volumes:
      - /etc/localtime:/etc/localtime:ro
      - ./pdns-admin-upload:/opt/powerdns-admin/upload
    networks:
      pdns:
        aliases:
          - pdns-admin-uwsgi
    depends_on:
      - mariadb
      - pdns-master

  pdns-admin-static:
    image: pschiffe/pdns-admin-static
    container_name: pdns-admin-static
    restart: unless-stopped
    ports:
      - '8989:80'
    volumes:
      - /etc/localtime:/etc/localtime:ro
    depends_on:
      - pdns-admin-uwsgi
    networks:
      - pdns
